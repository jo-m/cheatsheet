
% multicolumn table setup
\usepackage{supertabular}
\makeatletter
\let\mcnewpage=\newpage
\newcommand{\TrickSupertabularIntoMulticols}{
  \renewcommand\newpage{
    \if@firstcolumn
      \hrule width\linewidth height0pt
      \columnbreak
    \else
      \mcnewpage
    \fi
  }
}
\makeatother
